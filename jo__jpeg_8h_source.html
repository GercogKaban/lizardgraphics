47 static void jo_writeBits(FILE *fp, int *bitBuf, int *bitCnt, const unsigned short *bs) {
48  *bitCnt += bs[1];
49  *bitBuf |= bs[0] << (24 - *bitCnt);
50  while(*bitCnt >= 8) {
51  unsigned char c = (*bitBuf >> 16) & 255;
52  putc(c, fp);
53  if(c == 255) {
54  putc(0, fp);
55  }
56  *bitBuf <<= 8;
57  *bitCnt -= 8;
58  }
59 }

61 static void jo_DCT(float *d0, float *d1, float *d2, float *d3, float *d4, float *d5, float *d6, float *d7) {
62  float tmp0 = *d0 + *d7;
63  float tmp7 = *d0 - *d7;
64  float tmp1 = *d1 + *d6;
65  float tmp6 = *d1 - *d6;
66  float tmp2 = *d2 + *d5;
67  float tmp5 = *d2 - *d5;
68  float tmp3 = *d3 + *d4;
69  float tmp4 = *d3 - *d4;
70 
71  // Even part
72  float tmp10 = tmp0 + tmp3; // phase 2
73  float tmp13 = tmp0 - tmp3;
74  float tmp11 = tmp1 + tmp2;
75  float tmp12 = tmp1 - tmp2;
76 
77  *d0 = tmp10 + tmp11; // phase 3
78  *d4 = tmp10 - tmp11;
79 
80  float z1 = (tmp12 + tmp13) * 0.707106781f; // c4
81  *d2 = tmp13 + z1; // phase 5
82  *d6 = tmp13 - z1;
83 
84  // Odd part
85  tmp10 = tmp4 + tmp5; // phase 2
86  tmp11 = tmp5 + tmp6;
87  tmp12 = tmp6 + tmp7;
88 
89  // The rotator is modified from fig 4-8 to avoid extra negations.
90  float z5 = (tmp10 - tmp12) * 0.382683433f; // c6
91  float z2 = tmp10 * 0.541196100f + z5; // c2-c6
92  float z4 = tmp12 * 1.306562965f + z5; // c2+c6
93  float z3 = tmp11 * 0.707106781f; // c4
94 
95  float z11 = tmp7 + z3; // phase 5
96  float z13 = tmp7 - z3;
97 
98  *d5 = z13 + z2; // phase 6
99  *d3 = z13 - z2;
100  *d1 = z11 + z4;
101  *d7 = z11 - z4;
102 }

104 static void jo_calcBits(int val, unsigned short bits[2]) {
105  int tmp1 = val < 0 ? -val : val;
106  val = val < 0 ? val-1 : val;
107  bits[1] = 1;
108  while(tmp1 >>= 1) {
109  ++bits[1];
110  }
111  bits[0] = val & ((1<<bits[1])-1);
112 }

114 static int jo_processDU(FILE *fp, int *bitBuf, int *bitCnt, float *CDU, float *fdtbl, int DC, const unsigned short HTDC[256][2], const unsigned short HTAC[256][2]) {
115  const unsigned short EOB[2] = { HTAC[0x00][0], HTAC[0x00][1] };
116  const unsigned short M16zeroes[2] = { HTAC[0xF0][0], HTAC[0xF0][1] };
117  int dataOff, i, nrmarker;
118 
119  // DCT rows
120  for(dataOff=0; dataOff<64; dataOff+=8) {
121  jo_DCT(&CDU[dataOff], &CDU[dataOff+1], &CDU[dataOff+2], &CDU[dataOff+3], &CDU[dataOff+4], &CDU[dataOff+5], &CDU[dataOff+6], &CDU[dataOff+7]);
122  }
123  // DCT columns
124  for(dataOff=0; dataOff<8; ++dataOff) {
125  jo_DCT(&CDU[dataOff], &CDU[dataOff+8], &CDU[dataOff+16], &CDU[dataOff+24], &CDU[dataOff+32], &CDU[dataOff+40], &CDU[dataOff+48], &CDU[dataOff+56]);
126  }
127  // Quantize/descale/zigzag the coefficients
128  int DU[64];
129  for(i=0; i<64; ++i) {
130  float v = CDU[i]*fdtbl[i];
131  DU[s_jo_ZigZag[i]] = (int)(v < 0 ? ceilf(v - 0.5f) : floorf(v + 0.5f));
132  }
133 
134  // Encode DC
135  int diff = DU[0] - DC;
136  if (diff == 0) {
137  jo_writeBits(fp, bitBuf, bitCnt, HTDC[0]);
138  } else {
139  unsigned short bits[2];
140  jo_calcBits(diff, bits);
141  jo_writeBits(fp, bitBuf, bitCnt, HTDC[bits[1]]);
142  jo_writeBits(fp, bitBuf, bitCnt, bits);
143  }
144  // Encode ACs
145  int end0pos = 63;
146  for(; (end0pos>0)&&(DU[end0pos]==0); --end0pos) {
147  }
148  // end0pos = first element in reverse order !=0
149  if(end0pos == 0) {
150  jo_writeBits(fp, bitBuf, bitCnt, EOB);
151  return DU[0];
152  }
153  for(i = 1; i <= end0pos; ++i) {
154  int startpos = i;
155  for (; DU[i]==0 && i<=end0pos; ++i) {
156  }
157  int nrzeroes = i-startpos;
158  if ( nrzeroes >= 16 ) {
159  int lng = nrzeroes>>4;
160  for (nrmarker=1; nrmarker <= lng; ++nrmarker)
161  jo_writeBits(fp, bitBuf, bitCnt, M16zeroes);
162  nrzeroes &= 15;
163  }
164  unsigned short bits[2];
165  jo_calcBits(DU[i], bits);
166  jo_writeBits(fp, bitBuf, bitCnt, HTAC[(nrzeroes<<4)+bits[1]]);
167  jo_writeBits(fp, bitBuf, bitCnt, bits);
168  }
169  if(end0pos != 63) {
170  jo_writeBits(fp, bitBuf, bitCnt, EOB);
171  }
172  return DU[0];
173 }
