253 auto initRanges = [&](const glm::vec3* vec, VectorTypes vectorType)
254 {
255
256 const auto leftEndOffset = vectorType * coordsCount * SliderRanges::rangeEndsCount;
257 const auto rightEndOffset = leftEndOffset + coordsCount;
258
259 ranges.ranges[leftEndOffset] = vec->x - range;
260 ranges.ranges[rightEndOffset] = vec->x + range;
261 ranges.ranges[leftEndOffset + 1] = vec->y - range;
262 ranges.ranges[rightEndOffset + 1] = vec->y + range;
263 ranges.ranges[leftEndOffset + 2] = vec->z - range;
264 ranges.ranges[rightEndOffset + 2] = vec->z + range;
265 };

268 auto checkRanges = [&](const glm::vec3& vec, VectorTypes vectorType, size_t coordNum)
269 {
270 const auto leftOffset = vectorType * coordsCount * SliderRanges::rangeEndsCount;
271 const auto rightOffset = leftOffset + coordsCount;
272
273 if (vec[coordNum] <= ranges.ranges[leftOffset + coordNum])
274 {
275 ranges.ranges[leftOffset + coordNum] -= halfRange;
276 ranges.ranges[rightOffset + coordNum] -= halfRange;
277 }
278 else if (vec[coordNum] >= ranges.ranges[rightOffset + coordNum])
279 {
280 ranges.ranges[leftOffset + coordNum] += halfRange;
281 ranges.ranges[rightOffset + coordNum] += halfRange;
282 }
283 };
