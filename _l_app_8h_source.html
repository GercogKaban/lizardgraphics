555 template<typename T>
556 void createBuffer(T* buffer, VkBuffer& dst, VmaAllocation& allocation, VkBufferUsageFlags usage, size_t bufSize)
557 {
558 VkDeviceSize bufferSize;
559 VmaAllocation stagingAllocation;
560 VkBuffer stagingBuffer;
561
562 VmaStats stats;
563 vmaCalculateStats(allocator, &stats);
564
565 createBuffer(bufSize, VK_BUFFER_USAGE_TRANSFER_DST_BIT | usage,
566 VK_MEMORY_PROPERTY_DEVICE_LOCAL_BIT, dst, allocation);
567
568 auto unusedBytes = stats.memoryType[VK_MEMORY_PROPERTY_HOST_VISIBLE_BIT].unusedBytes;
569
570 if (!unusedBytes)
571 // 256 mb
572 unusedBytes = 268435456;
573
574 const auto batches = bufSize / unusedBytes;
575
576 auto stageBufTransfer = [&](VkDeviceSize offset)
577 {
578 createBuffer(bufferSize, VK_BUFFER_USAGE_TRANSFER_SRC_BIT,
579 VK_MEMORY_PROPERTY_HOST_VISIBLE_BIT | VK_MEMORY_PROPERTY_HOST_COHERENT_BIT,
580 stagingBuffer, stagingAllocation);
581
582 void* data;
583 vmaMapMemory(allocator, stagingAllocation, &data);
584 memcpy(data, buffer + (size_t)offset, bufferSize);
585 vmaUnmapMemory(allocator, stagingAllocation);
586
587 copyBuffer(stagingBuffer, dst, bufferSize, offset);
588
589 vmaDestroyBuffer(allocator, stagingBuffer, stagingAllocation);
590 };
591
592 for (size_t i = 0; i < batches; ++i)
593 {
594 bufferSize = unusedBytes;
595 bufSize -= bufferSize;
596 stageBufTransfer(i * unusedBytes);
597 }
598 if (bufSize)
599 {
600 bufferSize = bufSize;
601 stageBufTransfer(batches * unusedBytes);
602 }
603 }
