455  template<typename T>
456  void createBuffer(T* buffer, VkBuffer& dst, VmaAllocation& allocation, VkBufferUsageFlags usage, size_t bufSize)
457  {
458  VkDeviceSize bufferSize;
459  VmaAllocation stagingAllocation;
460  VkBuffer stagingBuffer;
461 
462  VmaStats stats;
463  vmaCalculateStats(allocator, &stats);
464 
465  createBuffer(bufSize, VK_BUFFER_USAGE_TRANSFER_DST_BIT | usage,
466  VK_MEMORY_PROPERTY_DEVICE_LOCAL_BIT, dst, allocation);
467 
468  auto unusedBytes = stats.memoryType[VK_MEMORY_PROPERTY_HOST_VISIBLE_BIT].unusedBytes;
469 
470  if (!unusedBytes)
471  // 256 mb
472  unusedBytes = 268435456;
473 
474  const auto batches = bufSize / unusedBytes;
475 
476  auto stageBufTransfer = [&](VkDeviceSize offset)
477  {
478  createBuffer(bufferSize, VK_BUFFER_USAGE_TRANSFER_SRC_BIT,
479  VK_MEMORY_PROPERTY_HOST_VISIBLE_BIT | VK_MEMORY_PROPERTY_HOST_COHERENT_BIT,
480  stagingBuffer, stagingAllocation);
481 
482  void* data;
483  vmaMapMemory(allocator, stagingAllocation, &data);
484  memcpy(data, buffer + (size_t)offset, bufferSize);
485  vmaUnmapMemory(allocator, stagingAllocation);
486 
487  copyBuffer(stagingBuffer, dst, bufferSize, offset);
488 
489  vmaDestroyBuffer(allocator, stagingBuffer, stagingAllocation);
490  };
491 
492  for (size_t i = 0; i < batches; ++i)
493  {
494  bufferSize = unusedBytes;
495  bufSize -= bufferSize;
496  stageBufTransfer(i* unusedBytes);
497  }
498  if (bufSize)
499  {
500  bufferSize = bufSize;
501  stageBufTransfer(batches * unusedBytes);
502  }
503  }
