311 static void tick()

313 while (true)

315 while (errors.size_())

318 errors.pop_();
319#ifndef NDEBUG
320 if (states & CONSOLE_DEBUG_LOG)

323 logToFile(err, firstCreationFlag);
324#else
325 if (states & CONSOLE_RELEASE_LOG)
326 println(err);
327 if (states & FILE_RELEASE_LOG)
328 logToFile(err);
329#endif
330 firstCreationFlag = ::std::ios_base::app;
331 }
332
333 if (stopLogging)
334 return;
335 }
336 }
