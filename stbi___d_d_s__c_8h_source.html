79 int stbi_convert_bit_range( int c, int from_bits, int to_bits )
80 {
81  int b = (1 << (from_bits - 1)) + c * ((1 << to_bits) - 1);
82  return (b + (b >> from_bits)) >> from_bits;
83 }

90 void stbi_decode_DXT1_block(
91  unsigned char uncompressed[16*4],
92  unsigned char compressed[8] )
93 {
94  int next_bit = 4*8;
95  int i, r, g, b;
96  int c0, c1;
97  unsigned char decode_colors[4*4];
98  // find the 2 primary colors
99  c0 = compressed[0] + (compressed[1] << 8);
100  c1 = compressed[2] + (compressed[3] << 8);
101  stbi_rgb_888_from_565( c0, &r, &g, &b );
102  decode_colors[0] = r;
103  decode_colors[1] = g;
104  decode_colors[2] = b;
105  decode_colors[3] = 255;
106  stbi_rgb_888_from_565( c1, &r, &g, &b );
107  decode_colors[4] = r;
108  decode_colors[5] = g;
109  decode_colors[6] = b;
110  decode_colors[7] = 255;
111  if( c0 > c1 )
112  {
113  // no alpha, 2 interpolated colors
114  decode_colors[8] = (2*decode_colors[0] + decode_colors[4]) / 3;
115  decode_colors[9] = (2*decode_colors[1] + decode_colors[5]) / 3;
116  decode_colors[10] = (2*decode_colors[2] + decode_colors[6]) / 3;
117  decode_colors[11] = 255;
118  decode_colors[12] = (decode_colors[0] + 2*decode_colors[4]) / 3;
119  decode_colors[13] = (decode_colors[1] + 2*decode_colors[5]) / 3;
120  decode_colors[14] = (decode_colors[2] + 2*decode_colors[6]) / 3;
121  decode_colors[15] = 255;
122  } else
123  {
124  // 1 interpolated color, alpha
125  decode_colors[8] = (decode_colors[0] + decode_colors[4]) / 2;
126  decode_colors[9] = (decode_colors[1] + decode_colors[5]) / 2;
127  decode_colors[10] = (decode_colors[2] + decode_colors[6]) / 2;
128  decode_colors[11] = 255;
129  decode_colors[12] = 0;
130  decode_colors[13] = 0;
131  decode_colors[14] = 0;
132  decode_colors[15] = 0;
133  }
134  // decode the block
135  for( i = 0; i < 16*4; i += 4 )
136  {
137  int idx = ((compressed[next_bit>>3] >> (next_bit & 7)) & 3) * 4;
138  next_bit += 2;
139  uncompressed[i+0] = decode_colors[idx+0];
140  uncompressed[i+1] = decode_colors[idx+1];
141  uncompressed[i+2] = decode_colors[idx+2];
142  uncompressed[i+3] = decode_colors[idx+3];
143  }
144  // done
145 }
146 void stbi_decode_DXT23_alpha_block(
147  unsigned char uncompressed[16*4],
148  unsigned char compressed[8] )
149 {
150  int i, next_bit = 0;
151  // each alpha value gets 4 bits
152  for( i = 3; i < 16*4; i += 4 )
153  {
154  uncompressed[i] = stbi_convert_bit_range(
155  (compressed[next_bit>>3] >> (next_bit&7)) & 15,
156  4, 8 );
157  next_bit += 4;
158  }
159 }
160 void stbi_decode_DXT45_alpha_block(
161  unsigned char uncompressed[16*4],
162  unsigned char compressed[8] )
163 {
164  int i, next_bit = 8*2;
165  unsigned char decode_alpha[8];
166  // each alpha value gets 3 bits, and the 1st 2 bytes are the range
167  decode_alpha[0] = compressed[0];
168  decode_alpha[1] = compressed[1];
169  if( decode_alpha[0] > decode_alpha[1] )
170  {
171  // 6 step intermediate
172  decode_alpha[2] = (6*decode_alpha[0] + 1*decode_alpha[1]) / 7;
173  decode_alpha[3] = (5*decode_alpha[0] + 2*decode_alpha[1]) / 7;
174  decode_alpha[4] = (4*decode_alpha[0] + 3*decode_alpha[1]) / 7;
175  decode_alpha[5] = (3*decode_alpha[0] + 4*decode_alpha[1]) / 7;
176  decode_alpha[6] = (2*decode_alpha[0] + 5*decode_alpha[1]) / 7;
177  decode_alpha[7] = (1*decode_alpha[0] + 6*decode_alpha[1]) / 7;
178  } else
179  {
180  // 4 step intermediate, pluss full and none
181  decode_alpha[2] = (4*decode_alpha[0] + 1*decode_alpha[1]) / 5;
182  decode_alpha[3] = (3*decode_alpha[0] + 2*decode_alpha[1]) / 5;
183  decode_alpha[4] = (2*decode_alpha[0] + 3*decode_alpha[1]) / 5;
184  decode_alpha[5] = (1*decode_alpha[0] + 4*decode_alpha[1]) / 5;
185  decode_alpha[6] = 0;
186  decode_alpha[7] = 255;
187  }
188  for( i = 3; i < 16*4; i += 4 )
189  {
190  int idx = 0, bit;
191  bit = (compressed[next_bit>>3] >> (next_bit&7)) & 1;
192  idx += bit << 0;
193  ++next_bit;
194  bit = (compressed[next_bit>>3] >> (next_bit&7)) & 1;
195  idx += bit << 1;
196  ++next_bit;
197  bit = (compressed[next_bit>>3] >> (next_bit&7)) & 1;
198  idx += bit << 2;
199  ++next_bit;
200  uncompressed[i] = decode_alpha[idx & 7];
201  }
202  // done
203 }
204 void stbi_decode_DXT_color_block(
205  unsigned char uncompressed[16*4],
206  unsigned char compressed[8] )
207 {
208  int next_bit = 4*8;
209  int i, r, g, b;
210  int c0, c1;
211  unsigned char decode_colors[4*3];
212  // find the 2 primary colors
213  c0 = compressed[0] + (compressed[1] << 8);
214  c1 = compressed[2] + (compressed[3] << 8);
215  stbi_rgb_888_from_565( c0, &r, &g, &b );
216  decode_colors[0] = r;
217  decode_colors[1] = g;
218  decode_colors[2] = b;
219  stbi_rgb_888_from_565( c1, &r, &g, &b );
220  decode_colors[3] = r;
221  decode_colors[4] = g;
222  decode_colors[5] = b;
223  // Like DXT1, but no choicees:
224  // no alpha, 2 interpolated colors
225  decode_colors[6] = (2*decode_colors[0] + decode_colors[3]) / 3;
226  decode_colors[7] = (2*decode_colors[1] + decode_colors[4]) / 3;
227  decode_colors[8] = (2*decode_colors[2] + decode_colors[5]) / 3;
228  decode_colors[9] = (decode_colors[0] + 2*decode_colors[3]) / 3;
229  decode_colors[10] = (decode_colors[1] + 2*decode_colors[4]) / 3;
230  decode_colors[11] = (decode_colors[2] + 2*decode_colors[5]) / 3;
231  // decode the block
232  for( i = 0; i < 16*4; i += 4 )
233  {
234  int idx = ((compressed[next_bit>>3] >> (next_bit & 7)) & 3) * 3;
235  next_bit += 2;
236  uncompressed[i+0] = decode_colors[idx+0];
237  uncompressed[i+1] = decode_colors[idx+1];
238  uncompressed[i+2] = decode_colors[idx+2];
239  }
240  // done
241 }

243 static int stbi__dds_info( stbi__context *s, int *x, int *y, int *comp, int *iscompressed ) {
244  int flags,is_compressed,has_alpha;
245  DDS_header header={0};
246 
247  if( sizeof( DDS_header ) != 128 )
248  {
249  return 0;
250  }
251 
252  stbi__getn( s, (stbi_uc*)(&header), 128 );
253 
254  if( header.dwMagic != (('D' << 0) | ('D' << 8) | ('S' << 16) | (' ' << 24)) ) {
255  stbi__rewind( s );
256  return 0;
257  }
258  if( header.dwSize != 124 ) {
259  stbi__rewind( s );
260  return 0;
261  }
262  flags = DDSD_CAPS | DDSD_HEIGHT | DDSD_WIDTH | DDSD_PIXELFORMAT;
263  if( (header.dwFlags & flags) != flags ) {
264  stbi__rewind( s );
265  return 0;
266  }
267  if( header.sPixelFormat.dwSize != 32 ) {
268  stbi__rewind( s );
269  return 0;
270  }
271  flags = DDPF_FOURCC | DDPF_RGB;
272  if( (header.sPixelFormat.dwFlags & flags) == 0 ) {
273  stbi__rewind( s );
274  return 0;
275  }
276  if( (header.sCaps.dwCaps1 & DDSCAPS_TEXTURE) == 0 ) {
277  stbi__rewind( s );
278  return 0;
279  }
280 
281  is_compressed = (header.sPixelFormat.dwFlags & DDPF_FOURCC) / DDPF_FOURCC;
282  has_alpha = (header.sPixelFormat.dwFlags & DDPF_ALPHAPIXELS) / DDPF_ALPHAPIXELS;
283 
284  *x = header.dwWidth;
285  *y = header.dwHeight;
286 
287  if ( !is_compressed ) {
288  *comp = 3;
289 
290  if ( has_alpha )
291  *comp = 4;
292  }
293  else
294  *comp = 4;
295 
296  if ( iscompressed )
297  *iscompressed = is_compressed;
298 
299  return 1;
300 }

339 static void * stbi__dds_load(stbi__context *s, int *x, int *y, int *comp, int req_comp)
340 {
341  // all variables go up front
342  stbi_uc *dds_data = NULL;
343  stbi_uc block[16*4];
344  stbi_uc compressed[8];
345  int flags, DXT_family;
346  int has_alpha, has_mipmap;
347  int is_compressed, cubemap_faces;
348  int block_pitch, num_blocks;
349  DDS_header header={0};
350  int i, sz, cf;
351  // load the header
352  if( sizeof( DDS_header ) != 128 )
353  {
354  return NULL;
355  }
356  stbi__getn( s, (stbi_uc*)(&header), 128 );
357  // and do some checking
358  if( header.dwMagic != (('D' << 0) | ('D' << 8) | ('S' << 16) | (' ' << 24)) ) return NULL;
359  if( header.dwSize != 124 ) return NULL;
360  flags = DDSD_CAPS | DDSD_HEIGHT | DDSD_WIDTH | DDSD_PIXELFORMAT;
361  if( (header.dwFlags & flags) != flags ) return NULL;
362  /* According to the MSDN spec, the dwFlags should contain
363  DDSD_LINEARSIZE if it's compressed, or DDSD_PITCH if
364  uncompressed. Some DDS writers do not conform to the
365  spec, so I need to make my reader more tolerant */
366  if( header.sPixelFormat.dwSize != 32 ) return NULL;
367  flags = DDPF_FOURCC | DDPF_RGB;
368  if( (header.sPixelFormat.dwFlags & flags) == 0 ) return NULL;
369  if( (header.sCaps.dwCaps1 & DDSCAPS_TEXTURE) == 0 ) return NULL;
370  // get the image data
371  s->img_x = header.dwWidth;
372  s->img_y = header.dwHeight;
373  s->img_n = 4;
374  is_compressed = (header.sPixelFormat.dwFlags & DDPF_FOURCC) / DDPF_FOURCC;
375  has_alpha = (header.sPixelFormat.dwFlags & DDPF_ALPHAPIXELS) / DDPF_ALPHAPIXELS;
376  has_mipmap = (header.sCaps.dwCaps1 & DDSCAPS_MIPMAP) && (header.dwMipMapCount > 1);
377  cubemap_faces = (header.sCaps.dwCaps2 & DDSCAPS2_CUBEMAP) / DDSCAPS2_CUBEMAP;
378  /* I need cubemaps to have square faces */
379  cubemap_faces &= (s->img_x == s->img_y);
380  cubemap_faces *= 5;
381  cubemap_faces += 1;
382  block_pitch = (s->img_x+3) >> 2;
383  num_blocks = block_pitch * ((s->img_y+3) >> 2);
384  /* let the user know what's going on */
385  *x = s->img_x;
386  *y = s->img_y;
387  *comp = s->img_n;
388  /* is this uncompressed? */
389  if( is_compressed )
390  {
391  /* compressed */
392  // note: header.sPixelFormat.dwFourCC is something like (('D'<<0)|('X'<<8)|('T'<<16)|('1'<<24))
393  DXT_family = 1 + (header.sPixelFormat.dwFourCC >> 24) - '1';
394  if( (DXT_family < 1) || (DXT_family > 5) ) return NULL;
395  /* check the expected size...oops, nevermind...
396  those non-compliant writers leave
397  dwPitchOrLinearSize == 0 */
398  // passed all the tests, get the RAM for decoding
399  sz = (s->img_x)*(s->img_y)*4*cubemap_faces;
400  dds_data = (unsigned char*)malloc( sz );
401  /* do this once for each face */
402  for( cf = 0; cf < cubemap_faces; ++ cf )
403  {
404  // now read and decode all the blocks
405  for( i = 0; i < num_blocks; ++i )
406  {
407  // where are we?
408  int bx, by, bw=4, bh=4;
409  int ref_x = 4 * (i % block_pitch);
410  int ref_y = 4 * (i / block_pitch);
411  // get the next block's worth of compressed data, and decompress it
412  if( DXT_family == 1 )
413  {
414  // DXT1
415  stbi__getn( s, compressed, 8 );
416  stbi_decode_DXT1_block( block, compressed );
417  } else if( DXT_family < 4 )
418  {
419  // DXT2/3
420  stbi__getn( s, compressed, 8 );
421  stbi_decode_DXT23_alpha_block ( block, compressed );
422  stbi__getn( s, compressed, 8 );
423  stbi_decode_DXT_color_block ( block, compressed );
424  } else
425  {
426  // DXT4/5
427  stbi__getn( s, compressed, 8 );
428  stbi_decode_DXT45_alpha_block ( block, compressed );
429  stbi__getn( s, compressed, 8 );
430  stbi_decode_DXT_color_block ( block, compressed );
431  }
432  // is this a partial block?
433  if( ref_x + 4 > (int)s->img_x )
434  {
435  bw = s->img_x - ref_x;
436  }
437  if( ref_y + 4 > (int)s->img_y )
438  {
439  bh = s->img_y - ref_y;
440  }
441  // now drop our decompressed data into the buffer
442  for( by = 0; by < bh; ++by )
443  {
444  int idx = 4*((ref_y+by+cf*s->img_x)*s->img_x + ref_x);
445  for( bx = 0; bx < bw*4; ++bx )
446  {
447 
448  dds_data[idx+bx] = block[by*16+bx];
449  }
450  }
451  }
452  /* done reading and decoding the main image...
453  stbi__skip MIPmaps if present */
454  if( has_mipmap )
455  {
456  int block_size = 16;
457  if( DXT_family == 1 )
458  {
459  block_size = 8;
460  }
461  for( i = 1; i < (int)header.dwMipMapCount; ++i )
462  {
463  int mx = s->img_x >> (i + 2);
464  int my = s->img_y >> (i + 2);
465  if( mx < 1 )
466  {
467  mx = 1;
468  }
469  if( my < 1 )
470  {
471  my = 1;
472  }
473  stbi__skip( s, mx*my*block_size );
474  }
475  }
476  }/* per cubemap face */
477  } else
478  {
479  /* uncompressed */
480  DXT_family = 0;
481  s->img_n = 3;
482  if( has_alpha )
483  {
484  s->img_n = 4;
485  }
486  *comp = s->img_n;
487  sz = s->img_x*s->img_y*s->img_n*cubemap_faces;
488  dds_data = (unsigned char*)malloc( sz );
489  /* do this once for each face */
490  for( cf = 0; cf < cubemap_faces; ++ cf )
491  {
492  /* read the main image for this face */
493  stbi__getn( s, &dds_data[cf*s->img_x*s->img_y*s->img_n], s->img_x*s->img_y*s->img_n );
494  /* done reading and decoding the main image...
495  stbi__skip MIPmaps if present */
496  if( has_mipmap )
497  {
498  for( i = 1; i < (int)header.dwMipMapCount; ++i )
499  {
500  int mx = s->img_x >> i;
501  int my = s->img_y >> i;
502  if( mx < 1 )
503  {
504  mx = 1;
505  }
506  if( my < 1 )
507  {
508  my = 1;
509  }
510  stbi__skip( s, mx*my*s->img_n );
511  }
512  }
513  }
514  /* data was BGR, I need it RGB */
515  for( i = 0; i < sz; i += s->img_n )
516  {
517  unsigned char temp = dds_data[i];
518  dds_data[i] = dds_data[i+2];
519  dds_data[i+2] = temp;
520  }
521  }
522  /* finished decompressing into RGBA,
523  adjust the y size if we have a cubemap
524  note: sz is already up to date */
525  s->img_y *= cubemap_faces;
526  *y = s->img_y;
527  // did the user want something else, or
528  // see if all the alpha values are 255 (i.e. no transparency)
529  has_alpha = 0;
530  if( s->img_n == 4)
531  {
532  for( i = 3; (i < sz) && (has_alpha == 0); i += 4 )
533  {
534  has_alpha |= (dds_data[i] < 255);
535  }
536  }
537  if( (req_comp <= 4) && (req_comp >= 1) )
538  {
539  // user has some requirements, meet them
540  if( req_comp != s->img_n )
541  {
542  dds_data = stbi__convert_format( dds_data, s->img_n, req_comp, s->img_x, s->img_y );
543  *comp = req_comp;
544  }
545  } else
546  {
547  // user had no requirements, only drop to RGB is no alpha
548  if( (has_alpha == 0) && (s->img_n == 4) )
549  {
550  dds_data = stbi__convert_format( dds_data, 4, 3, s->img_x, s->img_y );
551  *comp = 3;
552  }
553  }
554  // OK, done
555  return dds_data;
556 }
